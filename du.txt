#!/bin/bash
array=("m" "o" "t" "u" "w" "e" "t" "h" "f" "r" "s" "a" "s" "u")
read x
num=0
day=0
for(( i=0;i<${#array[@]};i++))
do
y=${array[i]}
if [ "$y"x = "$x"x ]
then
((num++))
day=$i
fi
((i++))
done
if [ $num -eq 1 ]
then
((day++))
((day++))
fi
if [ $num -eq 2 ]
then
read xx
for(( i=0;i<${#array[@]};i++))
do
y=${array[i]}
yy=${array[i+1]}
if [ "$y"x = "$x"x ]
then
if [ "$yy"x = "$xx"x ]
then
day=$i
((day++))
((day++))
fi
fi
done
fi
if [ $day -eq 2 ]
then
echo "星期一"
fi
if [ $day -eq 4 ]
then
echo "星期二"
fi
if [ $day -eq 6 ]
then
echo "星期三"
fi
if [ $day -eq 8 ]
then
echo "星期四"
fi
if [ $day -eq 10 ]
then
echo "星期五"
fi
if [ $day -eq 12 ]
then
echo "星期六"
fi
if [ $day -eq 14 ]
then
echo "星期日"
fi
